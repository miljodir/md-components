export const CopyButton = ({ text }) => {
  return (
    <button
      onClick={() => navigator.clipboard.writeText(text)}
      title="Copy to clipboard"
      style={{
        display: 'inline-flex',
        alignItems: 'center',
        gap: '0.5rem',
        padding: '0.25rem 0.5rem',
        background: 'transparent',
        border: '1px solid #ccc',
        borderRadius: '4px',
        cursor: 'pointer',
        fontSize: '0.875rem'
      }}
    >
      {text}
      <MdIconContentCopy style={{ width: '1rem', height: '1rem' }} />
    </button>
  );
};

# Typography

Design system typography tokens for font families, sizes, and line heights.

## Font Families

### Sofia Pro
Used for headings and display text.

<div style={{ fontFamily: 'var(--md-font-family-sofia-pro)', fontSize: '1.5rem', marginBottom: '1rem' }}>
  The quick brown fox jumps over the lazy dog
</div>

**CSS Variable:** <CopyButton text="--md-font-family-sofia-pro" />

### Open Sans
Used for body text and UI elements.

<div style={{ fontFamily: 'var(--md-font-family-open-sans)', fontSize: '1rem', marginBottom: '1rem' }}>
  The quick brown fox jumps over the lazy dog
</div>
