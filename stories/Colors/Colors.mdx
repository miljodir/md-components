export const CopyButton = ({ text }) => {
  return (
    <button
      onClick={() => navigator.clipboard.writeText(text)}
      title="Copy to clipboard"
      style={{
        display: 'inline-flex',
        alignItems: 'center',
        gap: '0.5rem',
        padding: '0.25rem 0.5rem',
        background: 'transparent',
        border: '1px solid #ccc',
        borderRadius: '4px',
        cursor: 'pointer',
        fontSize: '0.875rem'
      }}
    >
      {text}
      <MdIconContentCopy style={{ width: '1rem', height: '1rem' }} />
    </button>
  );
};

export const ColorSwatch = ({ color, name, cssVar }) => {
  return (
    <div style={{ marginBottom: '1rem' }}>

      <div style={{ fontSize: '0.75rem', color: '#666', marginBottom: '0.5rem' }}>{color}</div>
      <CopyButton text={cssVar} />

  );
};
